function updateInsights(){ const total = filtered.length; const ativos = filtered.filter(d => d.ativo).length; const afast = filtered.filter(d => d.afastado).length; const laudos = filtered.filter(d => d.possuiLaudo).length; const risco65 = filtered.filter(d => d.risco65).length; const ec = filtered.filter(d => d.ecInformado).length; const atuaEC = filtered.filter(d => d.atuaComEC).length; const capacidadeLiquida = Math.max(0, ativos - afast - atuaEC); const capacidadePct = total ? ((capacidadeLiquida / total) * 100) : 0; const topGerencia = groupCount(filtered, "gerencia") .filter(x => x[0] !== "Não informado")[0] || ["-",0]; const topBairro = groupCount( filtered.filter(d => !d.foraRio), "bairroRes" ) .filter(x => x[0] && x[0] !== "Não informado")[0] || ["-",0]; const foraRio = filtered.filter(d => d.foraRio).length; const mediaIdade = avg(filtered, "idade"); const vulnerabilidade = ((afast + laudos + risco65 + atuaEC) / total) * 100; let statusCapacidade = "Estável"; let corCapacidade = "#22d3ee"; if(capacidadePct < 55){ statusCapacidade = "Crítica"; corCapacidade = "#fb7185"; } else if(capacidadePct < 70){ statusCapacidade = "Pressão operacional"; corCapacidade = "#fbbf24"; } const insights = [ { titulo: "Capacidade operacional líquida", texto: `${fmt(capacidadeLiquida)} servidores permanecem operacionalmente disponíveis, representando ${capacidadePct.toFixed(1).replace(".",",")}% do efetivo filtrado.`, cor: corCapacidade }, { titulo: "Pressão operacional", texto: `${fmt(afast)} servidores apresentam indisponibilidade registrada, impactando diretamente a capacidade de resposta operacional.`, cor: "#fbbf24" }, { titulo: "Risco de continuidade", texto: `${fmt(risco65)} servidores possuem 65 anos ou mais, indicando pressão futura de reposição e sucessão operacional.`, cor: "#fb7185" }, { titulo: "Dependência estrutural de EC", texto: `${fmt(atuaEC)} servidores atuam via FUNCAO_EC, sugerindo dependência operacional acima da estrutura formal de cargos.`, cor: "#a78bfa" }, { titulo: "Vulnerabilidade operacional", texto: `O índice consolidado de vulnerabilidade operacional do recorte atual é de ${vulnerabilidade.toFixed(1).replace(".",",")}%, considerando afastamentos, laudos, risco etário e atuação por EC.`, cor: "#38bdf8" }, { titulo: "Polarização territorial", texto: `${fmt(foraRio)} servidores residem fora do município do Rio de Janeiro, ampliando pressão logística e deslocamentos operacionais.`, cor: "#22d3ee" }, { titulo: "Concentração territorial", texto: `${topBairro[0]} representa atualmente o principal polo residencial do efetivo filtrado.`, cor: "#34d399" }, { titulo: "Envelhecimento da força de trabalho", texto: `A média etária atual do recorte é de ${mediaIdade.toFixed(1).replace(".",",")} anos.`, cor: "#f97316" }, { titulo: "Maior concentração gerencial", texto: `${topGerencia[0]} concentra ${fmt(topGerencia[1])} servidores no recorte atual, representando um dos principais polos operacionais.`, cor: "#60a5fa" } ]; document.getElementById("autoInsights").innerHTML = insights.map(i => `
${escapeHtml(i.titulo)} ${escapeHtml(i.texto)}
`).join(""); }
